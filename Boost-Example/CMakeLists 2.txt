cmake_minimum_required(VERSION 3.25)
project(Boost-Example)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(BOOST_ENABLE_CMAKE ON)

include(FetchContent) 
FetchContent_Declare(
	boost                                            # name of the content
	GIT_REPOSITORY https://github.com/boostorg/boost  # the repository
	GIT_TAG boost-1.81.0                                  # the tag
)
FetchContent_MakeAvailable(Boost)

add_executable(Boost-Example main.cpp)
target_link_libraries(Boost-Example PRIVATE Boost::headers Boost::system Boost::filesystem)
